model{
  ## Likelihood
  for (i in 1:N){
    recommendation[i, 1:ncat] ~ dmulti(pi[i, 1:ncat], M)
    # make probabilities
    for (j in 1:ncat){
      log(e[i,j]) <- alpha[j]
      pi[i,j] <- e[i,j] / sum(e[i,1:ncat])
    }
  }
  alpha[1] <- 0
  for (k in 2:ncat){
    alpha[k] ~ dnorm(0, 0.001)
  }
}
